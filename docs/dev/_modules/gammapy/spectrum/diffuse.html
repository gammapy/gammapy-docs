from __future__ import absolute_import, division, print_function, unicode_literals
from astropy.units import Quantity

__all__ = ['diffuse_gamma_ray_flux']


def _power_law(E, N, k):
    E = Quantity(E, 'TeV')
    E0 = Quantity(1, 'TeV')
    N = Quantity(N, 'm^-2 s^-1 TeV^-1 sr^-1')
    flux = N * (E / E0) ** (-k)
    return flux
